> 最基础的用 map 来解决。时间复杂度 O(n)

<!--more-->
```shell
func twoSum(nums []int, target int) []int {
	m := map[int]int{}

	for index, num := range nums {
		if v, ok := m[num]; ok {
			return []int{v, index}
		} else {
			m[target-num] = index
		}
	}
	return nil
}
```
